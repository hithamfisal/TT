const TARGET_SHEET_NAME = 'TT-History';
const HEADERS = [
  'SN.',
  "TT's",
  'Site ID',
  'Site Name',
  'Managed Resource ',
  'Issues',
  'Severity',
  'Region',
  'Observation Date',
  'Observation Time',
  'Recovery Date',
  'Recovery Time',
  'Escalated for L3 Support Date',
  'Escalated for L3 Support Time',
  'Total Duration/Days/Hours',
  'Duration (hrs)',
  'NE Detail/Impacted Object',
  'Service Impaction Status',
  'No. of correlated Alarms',
  'Escalated to ',
  'Escalation Level',
  'Status',
  'Comments Date',
  'Comments-Feedback',
  'Maintenance person/Team',
  'Maintenance Contact Details',
  'Action Taken/RCA',
  'Action',
  'RCA',
];

function jsonOutput(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}

function normalizeName(value) {
  return String(value || '').toLowerCase().replace(/[^a-z0-9]/g, '');
}

function getTargetSpreadsheet(payload) {
  if (payload.spreadsheetId) return SpreadsheetApp.openById(payload.spreadsheetId);
  return SpreadsheetApp.getActiveSpreadsheet();
}

function getTargetSheet(spreadsheet, payload) {
  const sheets = spreadsheet.getSheets();
  const requestedGid =
    payload.sheetGid !== undefined && payload.sheetGid !== null && payload.sheetGid !== ''
      ? Number(payload.sheetGid)
      : null;
  if (requestedGid !== null && !Number.isNaN(requestedGid)) {
    const byGid = sheets.find((sheet) => sheet.getSheetId() === requestedGid);
    if (byGid) return byGid;
  }

  const requestedName = payload.sheetName || TARGET_SHEET_NAME;
  const exact = spreadsheet.getSheetByName(requestedName);
  if (exact) return exact;

  const normalizedTarget = normalizeName(requestedName);
  const normalized = sheets.find((sheet) => normalizeName(sheet.getName()) === normalizedTarget);
  if (normalized) return normalized;

  const likely = sheets.find((sheet) => normalizeName(sheet.getName()).includes('tthistory'));
  if (likely) return likely;

  return null;
}

function findFirstEmptyTicketRow(sheet) {
  const minRow = 2;
  const lastRow = Math.max(sheet.getLastRow(), minRow);
  const ticketValues = sheet.getRange(minRow, 2, lastRow - minRow + 1, 1).getDisplayValues();
  for (let i = 0; i < ticketValues.length; i += 1) {
    if (!String(ticketValues[i][0] || '').trim()) return minRow + i;
  }
  return lastRow + 1;
}

function readLastTicketRows(sheet, count) {
  const lastRow = Math.max(sheet.getLastRow(), 1);
  const startRow = Math.max(1, lastRow - count + 1);
  return sheet
    .getRange(startRow, 1, lastRow - startRow + 1, Math.min(HEADERS.length, sheet.getLastColumn()))
    .getDisplayValues()
    .map((row, index) => ({
      rowNumber: startRow + index,
      tt: row[1] || '',
      siteId: row[2] || '',
      siteName: row[3] || '',
    }))
    .filter((row) => row.tt || row.siteId || row.siteName);
}

function findTicketRow(sheet, tt) {
  const target = String(tt || '').trim();
  if (!target) return null;
  const lastRow = Math.max(sheet.getLastRow(), 2);
  const ticketValues = sheet.getRange(2, 2, lastRow - 1, 1).getDisplayValues();
  for (let i = ticketValues.length - 1; i >= 0; i -= 1) {
    if (String(ticketValues[i][0] || '').trim() === target) return i + 2;
  }
  return null;
}

function parseSerialNumber(value) {
  const match = String(value || '').match(/\d+/);
  return match ? Number(match[0]) : 0;
}

function getPreviousSerialNumber(sheet, beforeRow) {
  if (beforeRow <= 2) return 0;
  const serialValues = sheet.getRange(2, 1, beforeRow - 2, 1).getDisplayValues();
  for (let i = serialValues.length - 1; i >= 0; i -= 1) {
    const serial = parseSerialNumber(serialValues[i][0]);
    if (serial > 0) return serial;
  }
  return 0;
}

function ensureSheetDimensions(sheet, requiredRows, requiredColumns) {
  const maxRows = sheet.getMaxRows();
  if (maxRows < requiredRows) {
    sheet.insertRowsAfter(maxRows, requiredRows - maxRows);
  }
  const maxColumns = sheet.getMaxColumns();
  if (maxColumns < requiredColumns) {
    sheet.insertColumnsAfter(maxColumns, requiredColumns - maxColumns);
  }
}

function copyFormatFromPreviousRow(sheet, startRow, rowCount) {
  const sourceRow = Math.max(1, startRow - 1);
  const maxColumns = Math.max(HEADERS.length, sheet.getLastColumn());
  ensureSheetDimensions(sheet, startRow + rowCount - 1, maxColumns);
  const source = sheet.getRange(sourceRow, 1, 1, maxColumns);
  const target = sheet.getRange(startRow, 1, rowCount, maxColumns);
  source.copyTo(target, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
}

function doGet(e) {
  const payload =
    e && e.parameter && e.parameter.spreadsheetId
      ? {
          spreadsheetId: e.parameter.spreadsheetId,
          sheetGid: e.parameter.sheetGid,
          sheetName: e.parameter.sheetName,
        }
      : {};
  const spreadsheet = getTargetSpreadsheet(payload);
  const sheet = spreadsheet ? getTargetSheet(spreadsheet, payload) : null;
  const findTt = e && e.parameter ? e.parameter.findTt : '';
  return jsonOutput({
    ok: true,
    message: 'DMR manual ticket save web app is deployed.',
    spreadsheetName: spreadsheet ? spreadsheet.getName() : '',
    targetSheetName: sheet ? sheet.getName() : '',
    targetSheetGid: sheet ? sheet.getSheetId() : '',
    nextWriteRow: sheet ? findFirstEmptyTicketRow(sheet) : '',
    foundTicketRow: sheet && findTt ? findTicketRow(sheet, findTt) : '',
    lastTicketRows: sheet ? readLastTicketRows(sheet, 10) : [],
    sheets: spreadsheet
      ? spreadsheet.getSheets().map((sheet) => ({ name: sheet.getName(), gid: sheet.getSheetId() }))
      : [],
    receivedAt: new Date().toISOString(),
  });
}

function doPost(e) {
  try {
    const payloadText =
      (e && e.parameter && e.parameter.payload) ||
      (e && e.postData && e.postData.contents) ||
      '{}';
    const payload = JSON.parse(payloadText);
    const rows = Array.isArray(payload.rows) ? payload.rows : [];
    const spreadsheet = getTargetSpreadsheet(payload);
    const sheet = getTargetSheet(spreadsheet, payload);

    if (!sheet) {
      return jsonOutput({
        ok: false,
        error: 'Sheet not found',
        requestedSheetName: payload.sheetName || TARGET_SHEET_NAME,
        requestedSheetGid: payload.sheetGid || '',
        spreadsheetName: spreadsheet ? spreadsheet.getName() : '',
        sheets: spreadsheet ? spreadsheet.getSheets().map((s) => ({ name: s.getName(), gid: s.getSheetId() })) : [],
      });
    }
    if (!rows.length) {
      return jsonOutput({ ok: true, appended: 0, message: 'No rows received.', sheetName: sheet.getName() });
    }

    const lock = LockService.getScriptLock();
    lock.waitLock(10000);
    try {
      const updateItems = [];
      const appendRows = [];
      rows.forEach((row) => {
        const rowNumber = Number(row.__rowNumber || row.rowNumber || 0);
        if (rowNumber >= 2) updateItems.push({ rowNumber, row });
        else appendRows.push(row);
      });

      const updatedRows = updateItems.map((item) => {
        const values = HEADERS.map((header) => item.row[header] ?? '');
        const existingSerial = sheet.getRange(item.rowNumber, 1).getDisplayValue();
        values[0] = existingSerial || item.row['SN.'] || '';
        ensureSheetDimensions(sheet, item.rowNumber, HEADERS.length);
        sheet.getRange(item.rowNumber, 1, 1, HEADERS.length).setValues([values]);
        return { rowNumber: item.rowNumber, tt: values[1] || '', siteId: values[2] || '', siteName: values[3] || '' };
      });

      let startRow = '';
      let writtenRows = [];
      if (appendRows.length) {
        const values = appendRows.map((row) => HEADERS.map((header) => row[header] ?? ''));
        startRow = findFirstEmptyTicketRow(sheet);
        const previousSerial = getPreviousSerialNumber(sheet, startRow);
        values.forEach((row, index) => {
          row[0] = previousSerial + index + 1;
        });
        copyFormatFromPreviousRow(sheet, startRow, values.length);
        ensureSheetDimensions(sheet, startRow + values.length - 1, HEADERS.length);
        sheet.getRange(startRow, 1, values.length, HEADERS.length).setValues(values);
        writtenRows = sheet
          .getRange(startRow, 1, values.length, HEADERS.length)
          .getDisplayValues()
          .map((row, index) => ({
            rowNumber: startRow + index,
            tt: row[1] || '',
            siteId: row[2] || '',
            siteName: row[3] || '',
          }));
      }

      SpreadsheetApp.flush();
      return jsonOutput({
        ok: true,
        appended: appendRows.length,
        updated: updatedRows.length,
        sheetName: sheet.getName(),
        sheetGid: sheet.getSheetId(),
        startRow,
        writtenRows,
        updatedRows,
      });
    } finally {
      lock.releaseLock();
    }
  } catch (error) {
    return jsonOutput({ ok: false, error: String(error && error.message ? error.message : error) });
  }
}
